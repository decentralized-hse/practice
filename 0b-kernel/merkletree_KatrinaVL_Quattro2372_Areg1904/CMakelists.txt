cmake_minimum_required(VERSION 3.15)

project(SecureSync VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

find_package(OpenSSL REQUIRED)

# set(OPENSSL_ROOT_DIR /usr/local/opt/openssl)

add_executable(secure_sync src/main.cpp)

target_include_directories(secure_sync PRIVATE $${OPENSSL_INCLUDE_DIR})

target_link_libraries(secure_sync PRIVATE OpenSSL::Crypto)

if(CMAKE_CXX_COMPILER_ID STREQUAL "GNU" AND CMAKE_CXX_COMPILER_VERSION VERSION_LESS 9.0)
    target_link_libraries(secure_sync PRIVATE stdc++fs)
endif()

message(STATUS "Build setup ready. OpenSSL found at: $${OPENSSL_INCLUDE_DIR}")
